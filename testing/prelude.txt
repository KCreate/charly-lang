func test(comp1, comp2, title) {
    let result = comp1 == comp2;
    if (result) {
        print(title + " passed!");
    } else {
        print(title + " failed!");
    };

    # Return the result
    result;
};

func repeat(times, callback) {
    let i = 0;
    while (i < times) {
        callback(i);
        i = i + 1;
    };
};
